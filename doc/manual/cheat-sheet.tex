\section{Global Options}
\begin{tabularx}{\columnwidth}{@{}lX@{}}

\cmd{-v, --verbose} & Show operational details \\

\section{Setup}

\subsection{maki init}
Create a new catalog in the current directory.

\subsection{maki volume add \textit{LABEL PATH}}
Register a storage volume. \\
\opt{--purpose working|archive|backup|cloud}

\subsection{maki volume list}
Show registered volumes with online/offline status. \\
\opt{--json} \opt{--verbose}

\subsection{maki volume set-purpose \textit{LABEL PURPOSE}}
Change a volume's logical role.

\subsection{maki volume remove \textit{LABEL}}
Unregister a volume. \opt{--apply}

\subsection{maki volume combine \textit{SOURCE TARGET}}
Merge two volumes. \opt{--apply}

\section{Ingest \& Edit}

\subsection{maki import \textit{PATHS...}}
Import files into the catalog. \\
\opt{--volume} \opt{--auto-group} \opt{--smart} \\
\opt{--embed} \ai{} \opt{--describe} \opt{--dry-run} \\
\opt{--include GROUP} \opt{--skip GROUP}

\subsection{maki delete \textit{ASSET\_IDS...}}
Delete assets. \opt{--apply} \opt{--remove-files}

\subsection{maki tag \textit{ASSET\_ID TAGS...}}
Add or remove tags. \opt{--remove}

\subsection{maki edit \textit{ASSET\_ID}}
Edit metadata fields. \\
\opt{--name} \opt{--description} \opt{--rating N} \\
\opt{--label COLOR} \opt{--date} \opt{--clear-FIELD}

\subsection{maki group \textit{HASHES...}}
Group variants into one asset.

\subsection{maki split \textit{ASSET\_ID} --variant \textit{HASH}}
Split variants out of an asset.

\subsection{maki auto-group \textit{[QUERY]}}
Auto-group by filename stem. \opt{--apply}

\subsection{maki describe \textit{[QUERY]}}
Generate VLM image descriptions. \\
\opt{--asset ID} \opt{--volume} \opt{--model} \\
\opt{--mode describe|tags|both} \\
\opt{--apply} \opt{--force} \opt{--dry-run}

\subsection{maki auto-tag \textit{[QUERY]} \ai}
AI auto-tagging with SigLIP. \\
\opt{--asset} \opt{--volume} \opt{--threshold} \\
\opt{--labels FILE} \opt{--apply} \opt{--download}

\subsection{maki embed \textit{[QUERY]} \ai}
Generate image embeddings. \\
\opt{--asset} \opt{--volume} \opt{--force} \opt{--export}

\subsection{maki collection list / show / delete}

\subsection{maki stack create \textit{ASSET\_IDS...}}

\subsection{maki stack remove / pick / dissolve / list / show}
Stack management commands.

\subsection{maki stack from-tag \textit{PATTERN}}
Convert tag patterns to stacks. \opt{--apply} \opt{--remove-tags}

\subsection{maki faces detect / cluster / people / name \ai}

\cmd{f:1.4-2.8} & Aperture range \\
\cmd{width:4000+} & Min width (px) \\
\cmd{height:2000+} & Min height (px) \\
\cmd{copies:2+} & File copies \\
\cmd{variants:2+} & Variant count \\

\subsection{Date Filters}

\section{Core Concepts}

\textbf{Asset} — Logical media item. \textbf{Variant} — Concrete file (roles: Original, Alternate, Processed, Export, Sidecar). \textbf{Recipe} — XMP sidecar (tracked by location). \textbf{Volume} — Storage device (online/offline). \textbf{Stack} — Collapsed group in browse. \textbf{Collection} — Static album. \textbf{Saved Search} — Dynamic smart album.

\section{Key Workflows}

\subsection{Import \& Organize}
\begin{enumerate}[leftmargin=1.5em]
\item \cmd{maki init} + \cmd{maki volume add}
\item \cmd{maki import --smart /path}
\item \cmd{maki serve} — browse in web UI
\item Rate, tag, label via UI or CLI
\item \cmd{maki auto-tag ""} \ai — AI tagging
\end{enumerate}

\subsection{Maintenance Cycle}
\begin{enumerate}[leftmargin=1.5em]
\item \cmd{maki verify} — check integrity
\item \cmd{maki sync} — detect moved/missing
\item \cmd{maki sync-metadata} — bidirectional XMP
\item \cmd{maki cleanup --apply} — remove stale
\end{enumerate}

\subsection{Backup \& Archive}
\begin{enumerate}[leftmargin=1.5em]
\item \cmd{maki volume add Archive /mnt/nas --purpose backup}
\item \cmd{maki relocate --target Archive ""} — copy all
\item \cmd{maki backup-status} — check coverage
\item \cmd{maki dedup --apply} — remove same-volume dups
\end{enumerate}

\subsection{Export \& Deliver}
\begin{enumerate}[leftmargin=1.5em]
\item \cmd{maki export "rating:4+ tag:portfolio" ./out}
\item \cmd{maki export "collection:Client" ./out --zip}
\item \cmd{maki contact-sheet "rating:5" portfolio.pdf}
\end{enumerate}

\section{Output Formats}

\cmd{maki search --format \textit{PRESET|TEMPLATE}}

\begin{tabularx}{\columnwidth}{@{}lX@{}}
\cmd{ids} & One UUID per line \\
\cmd{short} & ID, name, type, format, date \\
\cmd{full} & + tags, description, rating \\
\cmd{json} & JSON array \\
\cmd{'\{id\}\textbackslash t\{name\}'} & Custom template \\
\end{tabularx}

Placeholders: \cmd{\{id\}}, \cmd{\{name\}}, \cmd{\{filename\}}, \cmd{\{type\}}, \cmd{\{format\}}, \cmd{\{tags\}}, \cmd{\{description\}}, \cmd{\{rating\}}, \cmd{\{label\}}, \cmd{\{date\}}, \cmd{\{size\}}
