\secthead{Principles}{rules that keep the tag catalogue coherent over years}

\renewcommand{\arraystretch}{1.15}
\begin{tabularx}{\textwidth}{@{}>{\RaggedRight\arraybackslash}X >{\RaggedRight\arraybackslash}X >{\RaggedRight\arraybackslash}X >{\RaggedRight\arraybackslash}X@{}}
  \textbf{\color{maki-nori}1. Tag in hierarchies.} Separator is \cmd{|}. Ancestors auto-expand: \cmd{subject|nature|landscape} also stores \cmd{subject|nature} and \cmd{subject}. &
  \textbf{\color{maki-nori}2. Generic lowercase, proper nouns keep case.} \cmd{concert} not \cmd{Concert}; \cmd{Peter Schneider} stays capitalised; \cmd{Germany} in English but \cmd{München} local. &
  \textbf{\color{maki-nori}3. One language.} Pick one and stick with it. Place names: English for countries, local from region down. &
  \textbf{\color{maki-nori}4. Facets are orthogonal.} Independent axes, not nested. Intersect in queries; don't mash into one tree. \\[10pt]
  \multicolumn{4}{@{}l@{}}{\color{maki-cream!70!maki-stone}\rule{\textwidth}{0.4pt}} \\[8pt]
  \textbf{\color{maki-nori}5. Tag what you see.} Primary tags describe visible content; context goes in secondary tags. &
  \textbf{\color{maki-nori}6. Don't duplicate metadata.} Camera, lens, date, GPS are EXIF; folders are path. Don't tag them. &
  \textbf{\color{maki-nori}7. Aim for 5\,--\,15 tags per image.} Fewer is poor discoverability; more is usually noise. &
  \textbf{\color{maki-nori}8. Separate workflow from content.} Ratings / color labels / saved searches for state. Tags for content. \\
\end{tabularx}

\secthead{Tag commands}{CLI and web UI share the same operations and the same semantics}

\renewcommand{\arraystretch}{1.15}
\noindent
\begin{minipage}[t]{0.50\textwidth}
  \small
  \begin{tabularx}{\textwidth}{@{}l >{\RaggedRight\arraybackslash}X@{}}
    \cmd{maki tag \textit{ID} foo bar}            & Add tags (ancestors auto-expanded). \\
    \cmd{maki tag \textit{ID} --remove foo}       & Remove tags (orphaned ancestors cleaned). \\
    \cmd{maki tag clear \textit{ID}}              & Remove all tags from an asset. \\
    \cmd{maki tag rename \textit{OLD} \textit{NEW} --apply} & Rename a tag everywhere. Cascades to descendants. \\
    \cmd{maki tag split \textit{OLD} \textit{A B...} --apply} & Replace one tag with multiple (variadic). \\
    \cmd{~~...\ --keep}                           & Keep source alongside (additive / copy). \\
    \cmd{maki tag expand-ancestors --apply}       & Backfill ancestor paths on existing tags. \\
    \cmd{maki tag export-vocabulary}              & Export the tag tree as \cmd{vocabulary.yaml}. \\
    \cmd{~~...\ --default}                        & Export the built-in starter vocabulary (ignores catalogue). \\
  \end{tabularx}
\end{minipage}\hfill
\begin{minipage}[t]{0.47\textwidth}
  \small
  \begin{tabularx}{\textwidth}{@{}l >{\RaggedRight\arraybackslash}X@{}}
    \cmd{tag:landscape}                           & Tag match at any level (default). \\
    \cmd{tag:=landscape}                          & Whole-path match (exact tag value). \\
    \cmd{tag:/landscape}                          & Leaf only at any level. \\
    \cmd{tag:\textasciicircum Landscape}          & Case-sensitive. \\
    \cmd{tag:|wed}                                & Prefix anchor (\cmd{wedding}, \cmd{wedding-...}). \\
    \cmd{-tag:rejected}                           & Exclude. \\
    \cmd{tag:a,b}                                 & Either \cmd{a} or \cmd{b}. \\
    \cmd{tag:a tag:b}                             & Both \cmd{a} and \cmd{b}. \\
    \cmd{event:jane person:Alice color:warm}      & Intersect three facets. \\
  \end{tabularx}\\[2pt]
  {\scriptsize\color{maki-stone}\itshape Same filters work in CLI \cmd{maki search} and in the web UI search box.}
\end{minipage}
